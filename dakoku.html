const STORE_IPS = { '東京銀座店': [], '東京新宿店': ['106.72.15.98'], '東京池袋店': [], '横浜駅前店': ['106.73.57.35'], '大阪梅田店': ['14.10.162.166'], '大阪なんば店': [], '上野公演部屋': [], '横浜公演部屋': ['106.73.57.35'], }; // 列インデックス（0始まり） // A=0 B=1 C=2(本名) D=3 E=4 F=5 G=6 H=7 I=8 J=9 K=10 L=11 M=12 N=13 // N=13(残業時間) O=14(残業手当) P=15(深夜時間) Q=16(深夜手当) R=17(通勤手当) S=18(GM手当) T=19(給与合計) // U=20(IP) V=21(何人目) W=22(打刻管理２) X=23(エラー) Y=24(IPエラー) const COL = { date: 0, // A staff: 1, // B realName: 2, // C ★本名 store: 3, // D shift: 4, // E gmTitle: 5, // F inTime: 6, // G 出勤打刻時刻 wageIn: 7, // H lastCust: 8, // I clockOut: 9, // J 退勤打刻時刻 ('退勤待ち' or 実時刻) wageOut: 10, // K hourly: 11, // L workMins: 12, // M normalPay: 13, // N overMins: 14, // O ★残業時間(分) 480分超過分 overPay: 15, // P ★残業手当(円) 超過分×時給×0.25 lateMins: 16, // Q 深夜勤務時間(分) latePay: 17, // R 深夜割増手当(円) transport: 18, // S allowance: 19, // T total: 20, // U ip: 21, // V storeCount:22, // W ★本日この店舗で何人目か multi: 23, // X (「打刻管理２」からの打刻です) errors: 24, // Y ipError: 25, // Z }; // ===================== // 日付を "YYYY/M/D" 形式に正規化（Date型・文字列型両対応・JST固定） // ===================== function normDate_(v) { if (!v) return ''; if (v instanceof Date) { // タイムゾーンずれ防止のためUTC+9（JST）で取得 const jst = new Date(v.getTime() + 9 * 60 * 60 * 1000); return jst.getUTCFullYear() + '/' + (jst.getUTCMonth() + 1) + '/' + jst.getUTCDate(); } return String(v); } // ===================== // 今日の日付を "YYYY/M/D" 形式で返す（JST固定） // ===================== function todayJST_() { const now = new Date(); const jst = new Date(now.getTime() + 9 * 60 * 60 * 1000); return jst.getUTCFullYear() + '/' + (jst.getUTCMonth() + 1) + '/' + jst.getUTCDate(); } // ===================== // シフトラベルからシフトキーを取得 // ===================== function shiftKey_(label) { if (!label) return null; if (label.includes('早番')) return '早番'; if (label.includes('遅番')) return '遅番'; if (label.includes('全日')) return '全日'; return null; } // シフト時間帯の重複マップ const SHIFT_OVERLAP = { '早番': ['早番', '全日'], '遅番': ['遅番', '全日'], '全日': ['早番', '遅番', '全日'], }; // ===================== // 重複チェック // ルール①：同日・同スタッフ・シフト時間帯重複 → エラー（店舗違っても・multi問わず） // ルール②：同日・同店舗・同multiフラグ・シフト時間帯重複 → 別スタッフでもエラー // GM公演は対象外 // selfRowIndex: 退勤時に自分自身の出勤行を除外するため（0始まりインデックス、不要なら-1） // ===================== function checkDuplicates_(allRows, dateStr, staff, store, shiftLabel, multiFlag, selfRowIndex) { const errors = []; const newKey = shiftKey_(shiftLabel); if (!newKey) return errors; // GM公演はスキップ for (let i = 1; i < allRows.length; i++) { if (i === selfRowIndex) continue; const row = allRows[i]; const tCell = String(row[COL.storeCount] || ''); // T列で日付判定 const rowStaff = String(row[COL.staff] || ''); const rowStore = String(row[COL.store] || ''); const rowShift = String(row[COL.shift] || ''); const rowMulti = String(row[COL.multi] || ''); if (!tCell.startsWith(dateStr)) continue; // T列テキストで日付フィルタ if (rowShift === 'GM公演') continue; const rowKey = shiftKey_(rowShift); if (!rowKey) continue; if (!SHIFT_OVERLAP[newKey] || !SHIFT_OVERLAP[newKey].includes(rowKey)) continue; // ルール①：同スタッフ重複 if (rowStaff === staff) { errors.push( '本日この店舗・同シフト時間帯に同スタッフの打刻があります（' + rowStore + '）' ); continue; } // ルール②：同店舗・同multiフラグの別スタッフ if (rowStore === store && rowMulti === multiFlag) { errors.push( '本日この店舗・同シフト時間帯に ' + rowStaff + ' の打刻があります' ); } } return errors; } // ===================== // 本日この店舗で何人目かをカウント // T列に書き込んだ日付テキスト（例："2026/5/2"）を読み取って同日判定 // A列の日付型に依存しないため確実 // ===================== function countStoreToday_(allRows, dateStr, store, shiftLabel, selfRowIndex) { const newKey = shiftKey_(shiftLabel); let count = 0; for (let i = 1; i < allRows.length; i++) { if (i === selfRowIndex) continue; const row = allRows[i]; const tCell = String(row[COL.storeCount] || ''); // T列："2026/5/2 1人目" 形式 const rowStore = String(row[COL.store] || ''); const rowShift = String(row[COL.shift] || ''); // T列の先頭が dateStr と一致するか（例："2026/5/2"） if (!tCell.startsWith(dateStr)) continue; if (rowStore !== store) continue; if (rowShift === 'GM公演') continue; if (newKey) { const rowKey = shiftKey_(rowShift); if (!rowKey) continue; if (!SHIFT_OVERLAP[newKey] || !SHIFT_OVERLAP[newKey].includes(rowKey)) continue; } count++; } return count; } // ===================== // メイン // ===================== function doPost(e) { try { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('打刻ログ'); const now = new Date(); const dateStr = todayJST_(); const store = data.store || ''; const clientIP = data.clientIP || ''; const multiFlag = data.multi ? '「打刻管理２」からの打刻です' : ''; const shiftLabel = data.shiftLabel || ''; function fmt(n) { return Number(n || 0).toLocaleString('ja-JP'); } // IPチェック let ipError = ''; const registeredIPs = STORE_IPS[store]; if (registeredIPs && registeredIPs.length > 0) { // IPが登録されている店舗のみチェック（空配列の店舗はスキップ） if (!registeredIPs.includes(clientIP)) { ipError = '未登録IP（' + clientIP + '）からの打刻です'; } } const allRows = sheet.getDataRange().getValues(); // ===================== // 出勤 // ===================== if (data.type === 'clockin') { const dupErrors = checkDuplicates_(allRows, dateStr, data.staff, store, shiftLabel, multiFlag, -1); const storeCount = countStoreToday_(allRows, dateStr, store, shiftLabel, -1); // 2人目以降かつ2人体制フラグなし → エラーも追加 if (storeCount >= 1 && !data.multi) { dupErrors.push('本日この店舗・同シフト時間帯に ' + storeCount + '人の打刻があります'); } // 退勤打刻なしを示すエラーを必ず追加（退勤時に消える） const allInErrors = [...(data.errors || []), ...dupErrors]; const inErrors = allInErrors.join(' / '); // 出勤打刻時点で仮のシフト終了時刻による暫定給与（calcPay）が送られてくる // 退勤打刻を忘れても給与が0円にならないようにするため、ここで全列を埋めておく const hasProvisional = !!data.calcPay; const provWorkMins = hasProvisional ? (data.calcPay.workMins || 0) : ''; const provNormalPay = hasProvisional ? fmt(data.calcPay.normal || 0) : ''; const provOverMins = hasProvisional ? (data.calcPay.overMins || 0) : ''; const provOverPay = hasProvisional ? fmt(data.calcPay.overPay || 0) : ''; const provLateMins = hasProvisional ? (data.calcPay.lateMins || 0) : ''; const provLatePay = hasProvisional ? fmt(data.calcPay.late || 0) : ''; const provAllowance = hasProvisional ? fmt(data.allowance || 0) : ''; const provTotal = hasProvisional ? fmt((data.calcPay.total || 0)) : ''; sheet.appendRow([ dateStr, data.staff, data.realName || '', store, shiftLabel, data.gmTitle || '', // A-F data.inTime, data.wageIn, // G-H '', '退勤待ち', data.provisionalWageOut || '', // I-K（J列は常に退勤待ち、K列は仮シフト終了時刻） fmt(data.hourlyWage || 1226), // L provWorkMins, provNormalPay, // M-N provOverMins, provOverPay, // O-P provLateMins, provLatePay, // Q-R 0, provAllowance, provTotal, // S-U（交通費は退勤時に確定） clientIP, // V dateStr + ' ' + (storeCount + 1) + '人目', // W ★日付+何人目 multiFlag, // X inErrors, // Y（'退勤打刻なし'を含む） ipError // Z ]); return ContentService.createTextOutput(JSON.stringify({result:'success'})).setMimeType(ContentService.MimeType.JSON); } // ===================== // 退勤 // ===================== if (data.type === 'clockout') { // 対応する出勤行を探す let targetRow = -1; for (let i = allRows.length - 1; i >= 1; i--) { if ( String(allRows[i][COL.staff] || '') === data.staff && String(allRows[i][COL.store] || '') === store && String(allRows[i][COL.clockOut] || '') === '退勤待ち' && String(allRows[i][COL.multi] || '') === multiFlag ) { targetRow = i + 1; break; } } const selfIndex = targetRow > 0 ? targetRow - 1 : -1; const dupErrors = checkDuplicates_(allRows, dateStr, data.staff, store, shiftLabel, multiFlag, selfIndex); const workMins = data.workMins || 0; const lateMins = data.lateMins || 0; const transport = data.transport || 0; const totalWithTransport = data.totalWithTransport || 0; const outErrors = [...(data.errors || []), ...dupErrors].join(' / '); if (targetRow > 0) { sheet.getRange(targetRow, COL.lastCust + 1).setValue(data.lastCustomer || ''); sheet.getRange(targetRow, COL.clockOut + 1).setValue(data.outTime); sheet.getRange(targetRow, COL.wageOut + 1).setValue(data.wageOut); sheet.getRange(targetRow, COL.workMins + 1).setValue(workMins); sheet.getRange(targetRow, COL.normalPay + 1).setValue(fmt(data.calcPay ? data.calcPay.normal : 0)); sheet.getRange(targetRow, COL.overMins + 1).setValue((data.calcPay && data.calcPay.overMins != null) ? data.calcPay.overMins : 0); sheet.getRange(targetRow, COL.overPay + 1).setValue(fmt((data.calcPay && data.calcPay.overPay != null) ? data.calcPay.overPay : 0)); sheet.getRange(targetRow, COL.lateMins + 1).setValue(lateMins); sheet.getRange(targetRow, COL.latePay + 1).setValue(fmt(data.calcPay ? data.calcPay.late : 0)); sheet.getRange(targetRow, COL.transport + 1).setValue(fmt(transport)); sheet.getRange(targetRow, COL.allowance + 1).setValue(fmt(data.allowance || 0)); sheet.getRange(targetRow, COL.total + 1).setValue(fmt(totalWithTransport)); // V列(errors)：既存エラーに退勤エラーを追記 const existV = String(sheet.getRange(targetRow, COL.errors + 1).getValue() || ''); sheet.getRange(targetRow, COL.errors + 1).setValue([existV, outErrors].filter(Boolean).join(' / ')); } else { // 出勤記録なし → 新規行追記 const storeCount = countStoreToday_(allRows, dateStr, store, shiftLabel, -1); sheet.appendRow([ dateStr, data.staff, data.realName || '', store, shiftLabel, data.gmTitle || '', data.inTime || '', data.wageIn, data.lastCustomer || '', data.outTime, data.wageOut, fmt(data.hourlyWage || 1226), workMins, fmt(data.calcPay ? data.calcPay.normal : 0), data.calcPay ? data.calcPay.overMins : 0, fmt(data.calcPay ? data.calcPay.overPay : 0), lateMins, fmt(data.calcPay ? data.calcPay.late : 0), fmt(transport), fmt(data.allowance || 0), fmt(totalWithTransport), clientIP, dateStr + ' ' + (storeCount + 1) + '人目', multiFlag, '出勤記録なし' + (outErrors ? ' / ' + outErrors : ''), ipError ]); } return ContentService.createTextOutput(JSON.stringify({result:'success'})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({result:'unknown'})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({result:'error', message:err.message})).setMimeType(ContentService.MimeType.JSON); } }
